COL1_KeyPressed
	LDR R4,=GPIOD_BASE
	LDR R7, [R6,#GPIO_ODR]
	BIC R7, R7, #(0XC3<<8)
	ORR R7, R7, #(0x43<<8)
	STR R4, [R3, #GPIO_ODR]
	LDR R6, =GPIOC_BASE
	LDR R7, [R6,#GPIO_IDR]
	TEQ R7, #0x0111
	BEQ SEND_ONE
	TEQ R7, #0x0111
	BEQ SEND_FOUR
	TEQ R7, #0x0111
	BEQ SEND_SEVEN
	TEQ R7, #0x0111
	BEQ SEND_STAR
